#include<stdio.h>
#include<unistd.h>
int main()
{
	int f = fork();
	int sum1 = 0;
	int sum2 = 0;
	int sum = 0;

	if(f == 0)
	{
		printf("Child process\n");
		for(int i=1; i<=5; i++)
		{
			printf("%d\n",i);
			sum1 = sum1 + i;
		}
		printf("Summation of child process: %d\n",sum1);
	}
	else
	{
		printf("Parent process\n");
		for(int j=6; j<=10; j++)
		{
			printf("%d\n",j);
			sum2 = sum2 + j;
		}
		printf("Summation of parent process: %d\n",sum2);
	}
	
	//sum = sum1 + sum2;

	return 0;
}
